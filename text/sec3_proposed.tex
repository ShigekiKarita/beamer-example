\begin{frame}{Main working steps}
\begin{enumerate}
    \item $I'$ : 現時刻における差分フレームのブロック表現
    \item $I_{sparse}$ : K-SVD による $I'$ の疎表現
    \item $I_{feature}$ : Gaussian Matrix による射影 $\Theta I_{sparse}$
    \item $\Lambda$ : K-Means による$I_{feature}$ の二値化
\end{enumerate}
